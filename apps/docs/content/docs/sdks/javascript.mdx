# JavaScript SDK

The official Sim JavaScript SDK provides a comprehensive, type-safe interface to the Sim API. Built with modern JavaScript and full TypeScript support, it includes automatic retries, error handling, and real-time collaboration features.

## Quick Start

### Initialize the Client

```javascript
import { SimClient } from '@sim/sdk';

// Initialize with API key
const sim = new SimClient({
  apiKey: 'your-api-key',
  baseUrl: 'https://api.sim.dev', // Optional, defaults to production
  timeout: 30000, // Optional, defaults to 30 seconds
  retries: 3 // Optional, defaults to 3 retries
});

// Or use environment variable SIM_API_KEY
const sim = new SimClient();
```

### Basic Workflow Operations

```javascript
// Create a new workflow
const workflow = await sim.workflows.create({
  name: 'Data Processing Pipeline',
  description: 'Processes incoming data',
  workspaceId: 'ws_abc123'
});

console.log('Created workflow:', workflow.id);

// Get workflow by ID
const retrieved = await sim.workflows.get(workflow.id);

// Update workflow
const updated = await sim.workflows.update(workflow.id, {
  description: 'Enhanced data processing pipeline'
});

// List workflows with filtering
const workflows = await sim.workflows.list({
  workspaceId: 'ws_abc123',
  status: 'active',
  limit: 20
});

// Delete workflow
await sim.workflows.delete(workflow.id);
```

## Configuration

### Client Options

```typescript
interface SimClientOptions {
  apiKey?: string;           // API key (or set SIM_API_KEY env var)
  baseUrl?: string;          // Base URL for API (default: https://api.sim.dev)
  timeout?: number;          // Request timeout in milliseconds (default: 30000)
  retries?: number;          // Number of retry attempts (default: 3)
  retryDelay?: number;       // Base delay between retries in ms (default: 1000)
  maxRetryDelay?: number;    // Maximum retry delay in ms (default: 10000)
  userAgent?: string;        // Custom user agent string
  debug?: boolean;           // Enable debug logging (default: false)
}
```

### Advanced Configuration

```javascript
const sim = new SimClient({
  apiKey: 'your-api-key',
  timeout: 60000,
  retries: 5,
  retryDelay: 2000,
  debug: process.env.NODE_ENV === 'development',
  
  // Custom retry logic
  shouldRetry: (error, attempt) => {
    if (attempt >= 5) return false;
    return [429, 500, 502, 503, 504].includes(error.status);
  },
  
  // Custom retry delay calculation
  getRetryDelay: (attempt, error) => {
    if (error.status === 429) {
      // Use server-provided retry-after header
      return error.retryAfter * 1000;
    }
    // Exponential backoff with jitter
    return Math.min(1000 * Math.pow(2, attempt) + Math.random() * 1000, 30000);
  }
});
```

## YAML Import/Export

### Import Workflows

```javascript
// Import from YAML string
const yamlContent = `
name: Sample Workflow
description: A simple workflow
blocks:
  - id: start
    type: trigger
    name: Start
`;

const imported = await sim.workflows.importYaml(yamlContent, {
  workspaceId: 'ws_abc123',
  validateOnly: false,
  overwriteExisting: false
});

// Import from file
import { readFile } from 'fs/promises';

const yamlFile = await readFile('./workflow.yaml', 'utf8');
const workflow = await sim.workflows.importYaml(yamlFile, {
  workspaceId: 'ws_abc123'
});

// Validate YAML before import
const validation = await sim.workflows.validateYaml(yamlContent);
if (!validation.isValid) {
  console.error('Validation errors:', validation.errors);
} else {
  const imported = await sim.workflows.importYaml(yamlContent);
}
```

## Error Handling

### Comprehensive Error Handling

```javascript
import { SimError, ValidationError, NetworkError, RateLimitError } from '@sim/sdk';

try {
  const workflow = await sim.workflows.create(workflowData);
} catch (error) {
  if (error instanceof ValidationError) {
    console.error('Validation failed:', error.validationErrors);
    error.validationErrors.forEach(err => {
      console.error(`- ${err.field}: ${err.message}`);
    });
  } else if (error instanceof RateLimitError) {
    console.error('Rate limited. Retry after:', error.retryAfter);
    // Wait and retry
    setTimeout(() => {
      // Retry the operation
    }, error.retryAfter * 1000);
  } else if (error instanceof NetworkError) {
    console.error('Network error:', error.message);
    // Handle network issues
  } else if (error instanceof SimError) {
    console.error('API error:', error.message);
    console.error('Request ID:', error.requestId);
  } else {
    console.error('Unexpected error:', error);
  }
}
```

### Custom Error Handling

```javascript
// Global error handler
sim.onError((error, context) => {
  console.error('SDK Error:', {
    error: error.message,
    context: context,
    requestId: error.requestId,
    timestamp: new Date().toISOString()
  });

  // Send to monitoring service
  if (error instanceof SimError && error.status >= 500) {
    monitoring.captureException(error, { 
      extra: { context, requestId: error.requestId }
    });
  }
});

// Retry configuration
sim.configure({
  retryCondition: (error, attempt) => {
    // Custom retry logic
    if (error instanceof RateLimitError) return attempt < 5;
    if (error instanceof NetworkError) return attempt < 3;
    return false;
  }
});
```

### Generic Types

```typescript
// Generic workflow type with custom metadata
interface WorkflowWithMetadata<T = Record<string, any>> extends Workflow {
  metadata: T;
}

interface ProductionMetadata {
  environment: 'production';
  deployedBy: string;
  deployedAt: string;
  version: string;
}

// Type-safe workflow operations
const deployToProduction = async (workflowId: string) => {
  const workflow = await sim.workflows.get<WorkflowWithMetadata<ProductionMetadata>>(workflowId);
  
  if (workflow.metadata.environment !== 'production') {
    throw new Error('Workflow not configured for production');
  }
  
  return await sim.workflows.deploy(workflowId, {
    environment: 'production'
  });
};
```

### Error Recovery

```javascript
// Implement circuit breaker pattern
class CircuitBreaker {
  constructor(threshold = 5, timeout = 60000) {
    this.threshold = threshold;
    this.timeout = timeout;
    this.failures = 0;
    this.lastFailTime = 0;
    this.state = 'closed'; // closed, open, half-open
  }

  async execute(fn) {
    if (this.state === 'open') {
      if (Date.now() - this.lastFailTime > this.timeout) {
        this.state = 'half-open';
      } else {
        throw new Error('Circuit breaker is open');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      throw error;
    }
  }

  onSuccess() {
    this.failures = 0;
    this.state = 'closed';
  }

  onFailure() {
    this.failures++;
    this.lastFailTime = Date.now();
    if (this.failures >= this.threshold) {
      this.state = 'open';
    }
  }
}

// Use circuit breaker with SDK
const circuitBreaker = new CircuitBreaker();

const safeApiCall = async () => {
  return await circuitBreaker.execute(async () => {
    return await sim.workflows.list();
  });
};
```
