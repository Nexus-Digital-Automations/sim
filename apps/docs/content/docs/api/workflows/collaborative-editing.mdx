# Collaborative Editing

The Sim API provides powerful collaborative editing features that allow multiple users to work on workflows simultaneously. With real-time synchronization, granular permissions, and conflict resolution, teams can build complex workflows together efficiently.

## Overview

Collaborative editing in Sim includes:

- **Real-time Updates** - See changes as they happen
- **User Presence** - Know who's currently editing
- **Granular Permissions** - Control who can view, edit, or manage workflows
- **Conflict Resolution** - Automatic handling of simultaneous edits
- **Version History** - Track all changes with author attribution
- **Live Comments** - Discuss changes in context

## Collaboration Permissions

## Real-time Features

### WebSocket Connection

Connect to real-time collaboration events:

```javascript
class WorkflowCollaboration {
  constructor(workflowId, apiKey) {
    this.workflowId = workflowId;
    this.apiKey = apiKey;
    this.ws = null;
    this.handlers = {};
  }

  connect() {
    const wsUrl = `wss://api.sim.dev/ws/workflows/${this.workflowId}/collaborate`;
    
    this.ws = new WebSocket(wsUrl, [], {
      headers: {
        'Authorization': `Bearer ${this.apiKey}`
      }
    });

    this.ws.onopen = () => {
      console.log('Connected to collaboration session');
      this.send('join', { user_id: 'current_user' });
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleMessage(data);
    };

    this.ws.onclose = () => {
      console.log('Collaboration session closed');
      setTimeout(() => this.connect(), 5000); // Reconnect
    };
  }

  handleMessage(data) {
    const handler = this.handlers[data.type];
    if (handler) {
      handler(data);
    }
  }

  // Register event handlers
  on(event, handler) {
    this.handlers[event] = handler;
  }

  // Send messages to collaboration session
  send(type, payload) {
    if (this.ws && this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({
        type,
        workflow_id: this.workflowId,
        timestamp: new Date().toISOString(),
        ...payload
      }));
    }
  }

  // Broadcast cursor position
  updateCursor(blockId, field, position) {
    this.send('cursor_update', {
      block_id: blockId,
      field: field,
      position: position
    });
  }

  // Broadcast content changes
  updateContent(blockId, field, content) {
    this.send('content_update', {
      block_id: blockId,
      field: field,
      content: content,
      change_id: this.generateChangeId()
    });
  }

  generateChangeId() {
    return `change_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
}

// Usage
const collab = new WorkflowCollaboration('wf_abc123', 'your-api-key');

// Handle real-time events
collab.on('user_joined', (data) => {
  console.log(`${data.user.display_name} joined the session`);
});

collab.on('user_left', (data) => {
  console.log(`${data.user.display_name} left the session`);
});

collab.on('cursor_update', (data) => {
  console.log(`${data.user.display_name} is editing ${data.block_id}.${data.field}`);
});

collab.on('content_update', (data) => {
  console.log(`Content updated by ${data.user.display_name}:`, data.content);
});

collab.connect();
```

### Collaboration Events

| Event Type | Description | Payload |
|------------|-------------|---------|
| `user_joined` | User joined collaboration session | `{user, timestamp}` |
| `user_left` | User left collaboration session | `{user, timestamp}` |
| `cursor_update` | Cursor position changed | `{user, block_id, field, position}` |
| `content_update` | Content was modified | `{user, block_id, field, content, change_id}` |
| `selection_update` | Text selection changed | `{user, block_id, field, selection}` |
| `comment_added` | Comment was added | `{user, comment, block_id, position}` |
| `workflow_saved` | Workflow was saved | `{user, version, timestamp}` |

## Conflict Resolution

### Operational Transform

Sim uses Operational Transform (OT) to handle simultaneous edits:

```javascript
class ConflictResolver {
  constructor() {
    this.pendingOperations = [];
    this.appliedOperations = [];
  }

  // Apply local operation and send to server
  applyLocalOperation(operation) {
    // Apply to local state
    this.applyOperation(operation);
    
    // Transform against pending operations
    const transformedOp = this.transformAgainstPending(operation);
    
    // Send to server
    this.sendToServer(transformedOp);
    
    // Track as pending
    this.pendingOperations.push(transformedOp);
  }

  // Handle remote operation from server
  handleRemoteOperation(operation) {
    // Transform against local pending operations
    let transformedOp = operation;
    
    for (const pendingOp of this.pendingOperations) {
      transformedOp = this.transformOperations(transformedOp, pendingOp);
    }
    
    // Apply transformed operation
    this.applyOperation(transformedOp);
    
    // Update pending operations
    this.updatePendingOperations(operation);
  }

  // Transform two operations for concurrent execution
  transformOperations(op1, op2) {
    // Implementation of operational transform logic
    // This handles cases like:
    // - Insertions at same position
    // - Deletions of same content
    // - Overlapping edits
    
    if (op1.type === 'insert' && op2.type === 'insert') {
      if (op1.position <= op2.position) {
        return {
          ...op2,
          position: op2.position + op1.content.length
        };
      }
    }
    
    if (op1.type === 'delete' && op2.type === 'insert') {
      if (op1.position <= op2.position) {
        return {
          ...op2,
          position: op2.position - op1.length
        };
      }
    }
    
    // Additional transform cases...
    return op2;
  }

  applyOperation(operation) {
    // Apply operation to local workflow state
    switch (operation.type) {
      case 'insert':
        this.insertText(operation.block_id, operation.field, 
                       operation.position, operation.content);
        break;
      case 'delete':
        this.deleteText(operation.block_id, operation.field,
                       operation.position, operation.length);
        break;
      case 'replace':
        this.replaceText(operation.block_id, operation.field,
                        operation.position, operation.length, operation.content);
        break;
    }
  }
}
```

### Conflict Resolution Strategies

<Callout type="info">
**Automatic Resolution**: Most conflicts are resolved automatically using operational transform. Manual resolution is only needed for complex structural changes.
</Callout>

1. **Text Edits** - Automatically merged using operational transform
2. **Block Configuration** - Last-write-wins with user notification
3. **Structure Changes** - Manual resolution required
4. **Concurrent Saves** - Automatic merge with conflict markers

### 1. Permission Management

```javascript
// Implement role-based permissions
const PERMISSIONS = {
  viewer: ['read', 'comment'],
  editor: ['read', 'comment', 'edit'],
  owner: ['read', 'comment', 'edit', 'share', 'delete'],
  admin: ['read', 'comment', 'edit', 'share', 'delete', 'manage']
};

function hasPermission(userRole, action) {
  return PERMISSIONS[userRole]?.includes(action) || false;
}

// Check before allowing actions
if (hasPermission(user.role, 'edit')) {
  await updateWorkflow(workflowId, changes);
} else {
  throw new Error('Insufficient permissions');
}
```

### 2. Conflict Prevention

```javascript
// Implement optimistic locking
class OptimisticLocking {
  constructor() {
    this.lockTimeout = 30000; // 30 seconds
    this.activeLocks = new Map();
  }

  async acquireLock(blockId, userId) {
    const existingLock = this.activeLocks.get(blockId);
    
    if (existingLock && existingLock.userId !== userId) {
      const timeLeft = existingLock.expiresAt - Date.now();
      if (timeLeft > 0) {
        throw new Error(`Block locked by ${existingLock.username} for ${Math.round(timeLeft/1000)}s`);
      }
    }

    // Acquire lock
    this.activeLocks.set(blockId, {
      userId,
      username: await this.getUserName(userId),
      expiresAt: Date.now() + this.lockTimeout
    });

    // Auto-release lock
    setTimeout(() => {
      this.releaseLock(blockId, userId);
    }, this.lockTimeout);
  }

  releaseLock(blockId, userId) {
    const lock = this.activeLocks.get(blockId);
    if (lock && lock.userId === userId) {
      this.activeLocks.delete(blockId);
    }
  }
}
```

### 3. User Presence

```javascript
// Track active users and their status
class PresenceManager {
  constructor(workflowId) {
    this.workflowId = workflowId;
    this.activeUsers = new Map();
  }

  updatePresence(userId, status) {
    this.activeUsers.set(userId, {
      ...status,
      lastSeen: Date.now()
    });

    // Broadcast to other users
    this.broadcastPresence(userId, status);
  }

  getActiveUsers() {
    const now = Date.now();
    const activeThreshold = 60000; // 1 minute
    
    return Array.from(this.activeUsers.entries())
      .filter(([_, presence]) => now - presence.lastSeen < activeThreshold)
      .map(([userId, presence]) => ({ userId, ...presence }));
  }

  broadcastPresence(userId, status) {
    // Send via WebSocket to all collaborators
    this.broadcast('presence_update', {
      user_id: userId,
      status: status,
      timestamp: new Date().toISOString()
    });
  }
}
```
