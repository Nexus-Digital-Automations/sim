# YAML Import/Export

The Sim API supports importing and exporting workflows in YAML format, enabling version control, programmatic workflow management, and easy sharing of workflow definitions. YAML format provides a human-readable, text-based representation of your workflows.

## Overview

YAML workflow operations include:

- **Import from YAML** - Create new workflows from YAML definitions
- **Update from YAML** - Update existing workflows with YAML
- **Validation Only** - Validate YAML without creating workflows
- **Layout Generation** - Automatically position blocks when importing
- **ID Preservation** - Maintain workflow and block IDs across imports

## Import Workflow from YAML

Create a new workflow from a YAML definition.

### Endpoint

```http
POST /api/workflows/yaml
```

### Request Body

```json
{
  "name": "string (required)",
  "description": "string (optional)",
  "color": "string (optional)",
  "workspaceId": "string (optional)",
  "folderId": "string | null (optional)",
  "yaml": "string (required)",
  "preserveIds": "boolean (default: false)",
  "overwriteExisting": "boolean (default: false)",
  "validateOnly": "boolean (default: false)"
}
```

### Parameters

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `name` | string | - | **Required.** Name for the new workflow |
| `description` | string | `""` | Optional description |
| `color` | string | `"#3972F6"` | Hex color code |
| `workspaceId` | string | - | Target workspace ID |
| `folderId` | string\|null | - | Target folder ID |
| `yaml` | string | - | **Required.** YAML workflow definition |
| `preserveIds` | boolean | `false` | Keep original workflow/block IDs |
| `overwriteExisting` | boolean | `false` | Overwrite existing workflow if ID matches |
| `validateOnly` | boolean | `false` | Only validate, don't create workflow |

### YAML Structure

A typical workflow YAML structure:

```yaml
version: "1.0"
metadata:
  name: "Email Processing Workflow"
  description: "Process incoming emails and extract data"
  color: "#10B981"

variables:
  customer_email:
    type: string
    description: "Customer email address"
  processed_count:
    type: number
    description: "Number of emails processed"
    default: 0

blocks:
  - id: start
    metadata:
      id: starter
      name: "Start"
    position: { x: 100, y: 100 }
    config:
      params:
        startWorkflow: "webhook"
        webhookPath: "/process-email"
    outputs:
      response:
        type: { input: "any" }
    
  - id: read_email
    metadata:
      id: gmail
      name: "Read Email"
    position: { x: 350, y: 100 }
    config:
      params:
        messageId: "{{webhook.messageId}}"
        markAsRead: true
    outputs:
      subject:
        type: { input: "string" }
      body:
        type: { input: "string" }
      sender:
        type: { input: "string" }
        
  - id: extract_data
    metadata:
      id: function
      name: "Extract Customer Data"
    position: { x: 600, y: 100 }
    config:
      params:
        code: |
          // Extract customer information from email
          const emailBody = inputs.body;
          const customerMatch = emailBody.match(/customer:\s*(\S+@\S+)/i);
          const customer_email = customerMatch ? customerMatch[1] : null;
          
          return { customer_email };
    outputs:
      customer_email:
        type: { input: "string" }

  - id: send_response
    metadata:
      id: gmail
      name: "Send Response"
    position: { x: 850, y: 100 }
    config:
      params:
        to: "{{extract_data.customer_email}}"
        subject: "Re: {{read_email.subject}}"
        body: "Thank you for your email. We'll process your request shortly."

connections:
  - source: start
    target: read_email
    sourceHandle: response
    targetHandle: input
    
  - source: read_email
    target: extract_data
    sourceHandle: body
    targetHandle: input
    
  - source: extract_data
    target: send_response
    sourceHandle: customer_email
    targetHandle: input

loops: {}
parallels: {}
```

## Update Workflow from YAML

Update an existing workflow with a new YAML definition.

### Endpoint

```http
PUT /api/workflows/yaml?id={workflowId}
```

### Query Parameters

| Parameter | Type | Description |
|-----------|------|-------------|
| `id` | string | **Required.** The workflow ID to update |

### Request Body

```json
{
  "yaml": "string (required)",
  "preserveMetadata": "boolean (default: true)",
  "createCheckpoint": "boolean (default: true)",
  "validateBeforeUpdate": "boolean (default: true)"
}
```

// Safe update with validation and error handling
async function safeUpdateWorkflow(workflowId, yamlContent) {
  try {
    // First, validate the YAML
    const validationResponse = await fetch('/api/workflows/yaml', {
      method: 'POST',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        name: 'temp',
        yaml: yamlContent,
        validateOnly: true
      })
    });

    const validation = await validationResponse.json();
    
    if (!validation.valid) {
      console.error('YAML validation failed:', validation.validationErrors);
      return { success: false, errors: validation.validationErrors };
    }

    console.log('YAML validation passed');

    // Now perform the update
    const updateResponse = await fetch(`/api/workflows/yaml?id=${workflowId}`, {
      method: 'PUT',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        yaml: yamlContent,
        preserveMetadata: true,
        createCheckpoint: true,
        validateBeforeUpdate: true
      })
    });

    if (!updateResponse.ok) {
      const error = await updateResponse.json();
      console.error('Update failed:', error);
      return { success: false, error: error.message };
    }

    const result = await updateResponse.json();
    console.log('Workflow updated successfully:', result.summary);
    
    return { success: true, result };

  } catch (error) {
    console.error('Update failed with exception:', error);
    return { success: false, error: error.message };
  }
}

// Usage
const updateResult = await safeUpdateWorkflow('wf_abc123', newYamlContent);
if (!updateResult.success) {
  console.error('Failed to update workflow:', updateResult.error || updateResult.errors);
}

## YAML Validation

Validate YAML workflow definitions without creating or updating workflows.

<Callout type="info">
Use validation to test YAML definitions before importing or updating workflows. This helps catch errors early and ensures smooth imports.
</Callout>

### Validation Process

The validation process checks for:

- **YAML Syntax** - Valid YAML format
- **Schema Compliance** - Correct workflow structure
- **Block Configuration** - Valid block types and parameters
- **Connection Validity** - Proper source/target references
- **Variable Dependencies** - Variable references and types
- **Tool Configuration** - Valid tool parameters and settings

### Advanced Validation Example

```javascript
class WorkflowYAMLValidator {
  constructor(apiKey) {
    this.apiKey = apiKey;
  }

  async validateYAML(yamlContent, options = {}) {
    const {
      checkBlockConfig = true,
      checkConnections = true,
      checkVariables = true,
      checkToolConfig = true
    } = options;

    try {
      const response = await fetch('/api/workflows/validate', {
        method: 'POST',
        headers: {
          'x-api-key': this.apiKey,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          format: 'yaml',
          workflow: yamlContent,
          validateBlockConfig: checkBlockConfig,
          validateConnections: checkConnections,
          validateToolConfig: checkToolConfig
        })
      });

      const result = await response.json();
      
      return {
        isValid: result.result.isValid,
        errors: result.result.errors,
        warnings: result.result.warnings,
        summary: result.result.summary,
        performance: result.result.performance
      };

    } catch (error) {
      return {
        isValid: false,
        errors: [{ 
          type: 'error', 
          code: 'VALIDATION_FAILED', 
          message: error.message 
        }],
        warnings: [],
        summary: null,
        performance: null
      };
    }
  }

  async validateWithDetailedReport(yamlContent) {
    const validation = await this.validateYAML(yamlContent);
    
    console.log('\n=== YAML Validation Report ===');
    console.log(`Status: ${validation.isValid ? '✅ Valid' : '❌ Invalid'}`);
    
    if (validation.summary) {
      console.log(`\nSummary:`);
      console.log(`- Blocks: ${validation.summary.totalBlocks}`);
      console.log(`- Connections: ${validation.summary.totalConnections}`);
      console.log(`- Loops: ${validation.summary.totalLoops}`);
      console.log(`- Parallels: ${validation.summary.totalParallels}`);
    }

    if (validation.errors.length > 0) {
      console.log(`\n❌ Errors (${validation.errors.length}):`);
      validation.errors.forEach(error => {
        const location = error.location 
          ? ` (${error.location.blockName || error.location.blockId})`
          : '';
        console.log(`  - ${error.message}${location}`);
      });
    }

    if (validation.warnings.length > 0) {
      console.log(`\n⚠️  Warnings (${validation.warnings.length}):`);
      validation.warnings.forEach(warning => {
        const location = warning.location 
          ? ` (${warning.location.blockName || warning.location.blockId})`
          : '';
        console.log(`  - ${warning.message}${location}`);
      });
    }

    if (validation.performance) {
      console.log(`\nPerformance: ${validation.performance.validationTimeMs}ms`);
    }

    console.log('===============================\n');
    
    return validation;
  }
}

// Usage
const validator = new WorkflowYAMLValidator('YOUR_API_KEY');
const validation = await validator.validateWithDetailedReport(yamlContent);

if (validation.isValid) {
  console.log('YAML is ready for import!');
} else {
  console.log('Please fix the errors before importing.');
}
```

    echo "$response" | jq -r '.validationErrors[]?.message // .error'
    validation_failed=1
  else
    echo "✅ $file is valid"
  fi
done

// workflow-manager.js - Manage workflows across branches
class WorkflowBranchManager {
  constructor(apiKey) {
    this.apiKey = apiKey;
  }

  async deployFromBranch(branch, environment) {
    console.log(`Deploying workflows from ${branch} to ${environment}`);
    
    // Get workflows from git
    const yamlFiles = await this.getYAMLFilesFromGit(branch);
    
    const results = [];
    
    for (const file of yamlFiles) {
      try {
        const result = await this.deployWorkflowFile(file, environment);
        results.push({ file: file.path, success: true, result });
        console.log(`✅ Deployed ${file.path}`);
      } catch (error) {
        results.push({ file: file.path, success: false, error: error.message });
        console.error(`❌ Failed to deploy ${file.path}:`, error.message);
      }
    }
    
    return results;
  }

  async deployWorkflowFile(file, environment) {
    const workspaceId = this.getWorkspaceForEnvironment(environment);
    
    const response = await fetch('/api/workflows/yaml', {
      method: 'POST',
      headers: {
        'x-api-key': this.apiKey,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        name: file.name,
        description: `Deployed from ${file.branch} branch`,
        workspaceId,
        yaml: file.content,
        overwriteExisting: true
      })
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || 'Deployment failed');
    }

    return await response.json();
  }

  async getYAMLFilesFromGit(branch) {
    // Implementation would integrate with git to read YAML files
    // This is a simplified example
    const { execSync } = require('child_process');
    
    const files = execSync(`git ls-tree -r ${branch} --name-only | grep -E '\\.(ya?ml)$'`, { encoding: 'utf8' })
      .trim()
      .split('\n')
      .filter(Boolean);
    
    return files.map(filePath => ({
      path: filePath,
      name: path.basename(filePath, path.extname(filePath)),
      content: execSync(`git show ${branch}:${filePath}`, { encoding: 'utf8' }),
      branch
    }));
  }

  getWorkspaceForEnvironment(environment) {
    const workspaces = {
      development: process.env.DEV_WORKSPACE_ID,
      staging: process.env.STAGING_WORKSPACE_ID,
      production: process.env.PROD_WORKSPACE_ID
    };
    
    return workspaces[environment] || workspaces.development;
  }
}

### Error Handling Example

```javascript
async function handleYAMLImportErrors(yamlContent, workflowName) {
  try {
    // First validate
    const validationResponse = await fetch('/api/workflows/validate', {
      method: 'POST',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        format: 'yaml',
        workflow: yamlContent,
        validateRequired: true,
        validateConnections: true,
        validateBlockConfig: true
      })
    });

    const validation = await validationResponse.json();
    
    if (!validation.result.isValid) {
      console.error('YAML Validation Failed:');
      
      // Group errors by type
      const errorsByType = validation.result.errors.reduce((acc, error) => {
        acc[error.code] = acc[error.code] || [];
        acc[error.code].push(error);
        return acc;
      }, {});

      // Handle different error types
      if (errorsByType.YAML_PARSE_ERROR) {
        console.error('YAML Syntax Errors:');
        errorsByType.YAML_PARSE_ERROR.forEach(error => {
          console.error(`  - ${error.message}`);
        });
        return { success: false, type: 'syntax', errors: errorsByType.YAML_PARSE_ERROR };
      }

      if (errorsByType.MISSING_REQUIRED_FIELD) {
        console.error('Missing Required Fields:');
        errorsByType.MISSING_REQUIRED_FIELD.forEach(error => {
          const location = error.location ? ` in block ${error.location.blockName}` : '';
          console.error(`  - ${error.message}${location}`);
        });
        return { success: false, type: 'required_fields', errors: errorsByType.MISSING_REQUIRED_FIELD };
      }

      if (errorsByType.CONNECTION_SOURCE_NOT_FOUND || errorsByType.CONNECTION_TARGET_NOT_FOUND) {
        console.error('Invalid Connections:');
        [...(errorsByType.CONNECTION_SOURCE_NOT_FOUND || []), ...(errorsByType.CONNECTION_TARGET_NOT_FOUND || [])]
          .forEach(error => {
            console.error(`  - ${error.message}`);
          });
        return { success: false, type: 'connections', errors: validation.result.errors };
      }

      // General validation failure
      return { success: false, type: 'validation', errors: validation.result.errors };
    }

    // Validation passed, attempt import
    const importResponse = await fetch('/api/workflows/yaml', {
      method: 'POST',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        name: workflowName,
        yaml: yamlContent
      })
    });

    if (!importResponse.ok) {
      const error = await importResponse.json();
      console.error('Import Failed:', error);
      return { success: false, type: 'import', error };
    }

    const result = await importResponse.json();
    return { success: true, workflow: result };

  } catch (error) {
    console.error('Unexpected error:', error);
    return { success: false, type: 'exception', error: error.message };
  }
}

// Usage with error handling
const result = await handleYAMLImportErrors(yamlContent, 'My Workflow');

if (!result.success) {
  switch (result.type) {
    case 'syntax':
      console.log('Please fix YAML syntax errors and try again');
      break;
    case 'required_fields':
      console.log('Please add missing required fields');
      break;
    case 'connections':
      console.log('Please fix connection references');
      break;
    case 'import':
      console.log('Import failed:', result.error.message);
      break;
    default:
      console.log('An unexpected error occurred');
  }
} else {
  console.log('Workflow imported successfully:', result.workflow.id);
}
```
