# Export Formats

The Sim API supports exporting workflows in multiple formats to meet different use cases. Whether you need YAML for version control, JSON for programmatic access, or ZIP archives for complete backups, we provide flexible export options.

## Supported Export Formats

| Format | Extension | Use Case | Security Filtering |
|--------|-----------|----------|-------------------|
| `yaml` | `.yaml` | Version control, CI/CD | ✅ Credentials removed |
| `json` | `.json` | Programmatic processing | ✅ Credentials removed |
| `zip` | `.zip` | Complete backup with assets | ✅ Credentials removed |

<Callout type="info">
**Security First**: All exports automatically remove sensitive credentials and connection details for security. Use the original workflow for execution.
</Callout>

## Single Workflow Export

### Export Individual Workflow

Export a single workflow in your preferred format:

<Tabs items={['JavaScript', 'Python', 'cURL']}>
<Tab value="JavaScript">
```javascript
// Export single workflow as YAML
async function exportWorkflow(workflowId, format = 'yaml') {
  const response = await fetch(`/api/workflows/${workflowId}/export`, {
    method: 'GET',
    headers: {
      'x-api-key': 'your-api-key',
      'Accept': getAcceptHeader(format)
    }
  });

  if (!response.ok) {
    throw new Error(`Export failed: ${response.status}`);
  }

  return await response.blob();
}

function getAcceptHeader(format) {
  const headers = {
    yaml: 'application/x-yaml',
    json: 'application/json',
    zip: 'application/zip'
  };
  return headers[format] || 'application/x-yaml';
}

// Usage examples
try {
  // Export as YAML
  const yamlBlob = await exportWorkflow('wf_abc123', 'yaml');
  
  // Export as JSON
  const jsonBlob = await exportWorkflow('wf_abc123', 'json');
  
  // Export as ZIP
  const zipBlob = await exportWorkflow('wf_abc123', 'zip');
  
  console.log('Workflows exported successfully');
} catch (error) {
  console.error('Export failed:', error.message);
}
```

```python
import requests
import io

def export_workflow(workflow_id, format='yaml', api_key=None):
    """Export a single workflow in specified format."""
    
    headers = {
        'x-api-key': api_key,
        'Accept': get_accept_header(format)
    }
    
    url = f"https://api.sim.dev/api/workflows/{workflow_id}/export"
    
    response = requests.get(url, headers=headers)
    response.raise_for_status()
    
    return response.content

def get_accept_header(format):
    """Get appropriate Accept header for format."""
    headers = {
        'yaml': 'application/x-yaml',
        'json': 'application/json', 
        'zip': 'application/zip'
    }
    return headers.get(format, 'application/x-yaml')

def save_export(content, filename):
    """Save exported content to file."""
    with open(filename, 'wb') as f:
        f.write(content)

# Usage examples
try:
    # Export as YAML
    yaml_content = export_workflow('wf_abc123', 'yaml', 'your-api-key')
    save_export(yaml_content, 'workflow.yaml')
    
    # Export as JSON
    json_content = export_workflow('wf_abc123', 'json', 'your-api-key')
    save_export(json_content, 'workflow.json')
    
    # Export as ZIP
    zip_content = export_workflow('wf_abc123', 'zip', 'your-api-key')
    save_export(zip_content, 'workflow.zip')
    
    print("Workflows exported successfully")
    
except requests.exceptions.RequestException as e:
    print(f"Export failed: {e}")
```

## Bulk Export

### Export Multiple Workflows

Export multiple workflows at once as a ZIP archive:

<Tabs items={['JavaScript', 'Python', 'cURL']}>
<Tab value="JavaScript">
```javascript
// Bulk export workflows
async function bulkExportWorkflows(workflowIds, format = 'zip') {
  const response = await fetch('/api/workflows/export', {
    method: 'POST',
    headers: {
      'x-api-key': 'your-api-key',
      'Content-Type': 'application/json',
      'Accept': format === 'zip' ? 'application/zip' : 'application/json'
    },
    body: JSON.stringify({
      workflow_ids: workflowIds,
      format: format,
      include_metadata: true,
      security_filter: true
    })
  });

  if (!response.ok) {
    throw new Error(`Bulk export failed: ${response.status}`);
  }

  return await response.blob();
}

// Export all workflows in workspace
async function exportAllWorkflows(workspaceId) {
  const response = await fetch('/api/workflows/export', {
    method: 'POST',
    headers: {
      'x-api-key': 'your-api-key',
      'Content-Type': 'application/json',
      'Accept': 'application/zip'
    },
    body: JSON.stringify({
      workspace_id: workspaceId,
      format: 'zip',
      include_metadata: true,
      include_version_history: false
    })
  });

  if (!response.ok) {
    throw new Error(`Workspace export failed: ${response.status}`);
  }

  return await response.blob();
}

// Usage
try {
  // Export specific workflows
  const specificExport = await bulkExportWorkflows([
    'wf_abc123',
    'wf_def456', 
    'wf_ghi789'
  ]);
  
  // Export all workspace workflows
  const workspaceExport = await exportAllWorkflows('ws_abc123');
  
  console.log('Bulk exports completed');
} catch (error) {
  console.error('Bulk export failed:', error.message);
}
```

```python
import requests
import json

def bulk_export_workflows(workflow_ids, format='zip', api_key=None):
    """Export multiple workflows as ZIP archive."""
    
    headers = {
        'x-api-key': api_key,
        'Content-Type': 'application/json',
        'Accept': 'application/zip' if format == 'zip' else 'application/json'
    }
    
    payload = {
        'workflow_ids': workflow_ids,
        'format': format,
        'include_metadata': True,
        'security_filter': True
    }
    
    response = requests.post(
        'https://api.sim.dev/api/workflows/export',
        headers=headers,
        json=payload
    )
    response.raise_for_status()
    
    return response.content

def export_workspace_workflows(workspace_id, api_key=None):
    """Export all workflows from a workspace."""
    
    headers = {
        'x-api-key': api_key,
        'Content-Type': 'application/json',
        'Accept': 'application/zip'
    }
    
    payload = {
        'workspace_id': workspace_id,
        'format': 'zip',
        'include_metadata': True,
        'include_version_history': False
    }
    
    response = requests.post(
        'https://api.sim.dev/api/workflows/export',
        headers=headers,
        json=payload
    )
    response.raise_for_status()
    
    return response.content

# Usage examples
try:
    # Export specific workflows
    bulk_content = bulk_export_workflows(
        ['wf_abc123', 'wf_def456', 'wf_ghi789'],
        'zip',
        'your-api-key'
    )
    
    with open('bulk_workflows.zip', 'wb') as f:
        f.write(bulk_content)
    
    # Export entire workspace
    workspace_content = export_workspace_workflows('ws_abc123', 'your-api-key')
    
    with open('workspace_export.zip', 'wb') as f:
        f.write(workspace_content)
    
    print("Bulk exports completed successfully")
    
except requests.exceptions.RequestException as e:
    print(f"Bulk export failed: {e}")
```

## Export Options

### Request Parameters

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `format` | string | `yaml` | Export format: `yaml`, `json`, or `zip` |
| `include_metadata` | boolean | `true` | Include creation/modification timestamps |
| `include_version_history` | boolean | `false` | Include version history (ZIP only) |
| `security_filter` | boolean | `true` | Remove credentials and sensitive data |
| `compress` | boolean | `true` | Enable compression for ZIP format |

### Security Filtering

<Callout type="warning">
**Important**: Security filtering is enabled by default and cannot be disabled via API. This ensures credentials and sensitive connection details are never exposed in exports.
</Callout>

**Filtered Data**:
- API keys and authentication tokens
- Database passwords and connection strings  
- OAuth client secrets
- Private certificates and keys
- Internal service endpoints

**Preserved Data**:
- Workflow structure and logic
- Block configurations (non-sensitive)
- Connection relationships
- Metadata and timestamps
- Block positions and UI layout

## ZIP Archive Structure

When exporting as ZIP, the archive contains:

```
workflow_export.zip
├── metadata.json          # Export information
├── workflows/
│   ├── workflow_1.yaml    # Individual workflow files
│   ├── workflow_2.yaml
│   └── workflow_3.yaml
├── assets/                # Referenced assets (if any)
│   ├── images/
│   └── documents/
└── manifest.json          # Archive contents listing
```

## Error Handling

### Common Export Errors

```javascript
async function handleExportErrors(workflowId) {
  try {
    const export = await exportWorkflow(workflowId);
    return export;
  } catch (error) {
    switch (error.status) {
      case 404:
        console.error('Workflow not found or no access');
        break;
      case 413:
        console.error('Export size too large - try bulk export');
        break;
      case 429:
        console.error('Rate limited - wait before retrying');
        break;
      case 503:
        console.error('Export service temporarily unavailable');
        break;
      default:
        console.error('Export failed:', error.message);
    }
    throw error;
  }
}
```

## Best Practices

### 1. Export Organization

```javascript
// Organize exports with naming conventions
function generateExportFilename(workflow, format) {
  const timestamp = new Date().toISOString().split('T')[0];
  const safeName = workflow.name.toLowerCase().replace(/[^a-z0-9]/g, '_');
  return `${timestamp}_${safeName}.${format}`;
}

// Example: "2024-01-15_data_processing_pipeline.yaml"
```

### 2. Batch Processing

```javascript
// Process large exports in batches
async function exportWorkspaceBatched(workspaceId, batchSize = 10) {
  const workflows = await getWorkspaceWorkflows(workspaceId);
  const batches = chunkArray(workflows, batchSize);
  
  const exports = [];
  for (const batch of batches) {
    const batchExports = await Promise.all(
      batch.map(wf => exportWorkflow(wf.id, 'yaml'))
    );
    exports.push(...batchExports);
    
    // Rate limiting pause
    await new Promise(resolve => setTimeout(resolve, 1000));
  }
  
  return exports;
}
```

### 3. Integrity Verification

```javascript
// Verify export integrity
async function verifyExport(originalWorkflow, exportedContent) {
  const parsed = yaml.parse(exportedContent);
  
  // Check essential structure
  const checks = [
    parsed.name === originalWorkflow.name,
    parsed.blocks?.length === originalWorkflow.blocks?.length,
    parsed.connections?.length === originalWorkflow.connections?.length
  ];
  
  return checks.every(check => check === true);
}
```
