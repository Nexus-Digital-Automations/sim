# Bulk Operations

The Sim API provides powerful bulk operations that allow you to manage multiple workflows simultaneously. This is essential for administrative tasks, data migration, batch processing, and organizational workflows.

## Overview

Bulk operations support the following actions:
- **Delete** - Remove multiple workflows at once
- **Move** - Relocate workflows to different folders or workspaces
- **Copy** - Duplicate workflows with customization options
- **Deploy** - Deploy multiple workflows simultaneously
- **Undeploy** - Undeploy multiple workflows
- **Tag** - Apply tags to multiple workflows for organization

## Endpoint

```http
PATCH /api/workflows
```

## Request Format

### Request Body Schema

```json
{
  "operation": "delete" | "move" | "copy" | "deploy" | "undeploy" | "tag",
  "workflowIds": ["string"],
  "targetFolderId": "string | null", // For move/copy operations
  "targetWorkspaceId": "string",     // For move/copy operations
  "tags": ["string"],                // For tag operations
  "preserveCollaborators": boolean   // For copy operations (default: true)
}
```

### Parameters

| Parameter | Type | Required | Description |
|-----------|------|----------|-------------|
| `operation` | string | ✅ | The bulk operation to perform |
| `workflowIds` | string[] | ✅ | Array of workflow IDs (1-100 workflows) |
| `targetFolderId` | string\|null | ⚠️ | Required for move/copy operations |
| `targetWorkspaceId` | string | ⚠️ | Required for move/copy operations |
| `tags` | string[] | ⚠️ | Required for tag operations |
| `preserveCollaborators` | boolean | ❌ | For copy operations (default: true) |

<Callout type="info">
**Limits**: You can perform bulk operations on up to 100 workflows per request. For larger operations, split into multiple requests.
</Callout>

## Bulk Delete

Remove multiple workflows permanently.

<Callout type="warn">
**Warning**: Bulk delete operations are irreversible. Ensure you have backups or exports if needed.
</Callout>

### Example

When you lack permission for some workflows:

```json
{
  "error": "Access denied for some workflows",

    { "id": "wf_456", "name": "Protected Workflow" }
```

### Error Handling Example

```javascript
async function safeBulkOperation(operation, workflowIds, options = {}) {
  try {
    const response = await fetch('/api/workflows', {
      method: 'PATCH',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operation,
        workflowIds,
        ...options
      })
    });

    if (!response.ok) {
      const error = await response.json();
      
      if (response.status === 403 && error.deniedWorkflows) {
        console.warn('Access denied for some workflows:', error.deniedWorkflows);
        
        // Retry with only accessible workflows
        const accessibleIds = workflowIds.filter(id => 
          !error.deniedWorkflows.some(denied => denied.id === id)
        );
        
        if (accessibleIds.length > 0) {
          console.log(`Retrying with ${accessibleIds.length} accessible workflows`);
          return safeBulkOperation(operation, accessibleIds, options);
        }
      }
      
      throw new Error(error.message || 'Bulk operation failed');
    }

    const result = await response.json();
    
    // Log results
    console.log(`Bulk ${operation} summary:`, result.summary);
    
    if (result.summary.failed > 0) {
      const failures = result.results.filter(r => r.status === 'error');
      console.warn('Some operations failed:', failures);
    }
    
    return result;
  } catch (error) {
    console.error(`Bulk ${operation} error:`, error);
    throw error;
  }
}

// Usage
try {
  const result = await safeBulkOperation('delete', ['wf_123', 'wf_456']);
  console.log(`Successfully deleted ${result.summary.successful} workflows`);
} catch (error) {
  console.error('Bulk delete failed:', error);
}
```

## Advanced Bulk Operations

### Batch Processing Large Sets

For operations on more than 100 workflows, split into batches:

```javascript
class BulkOperationManager {
  constructor(apiKey, batchSize = 100) {
    this.apiKey = apiKey;
    this.batchSize = batchSize;
  }

  async executeBulkOperation(operation, workflowIds, options = {}) {
    const batches = this.createBatches(workflowIds);
    const results = [];
    
    console.log(`Processing ${workflowIds.length} workflows in ${batches.length} batches`);
    
    for (let i = 0; i < batches.length; i++) {
      const batch = batches[i];
      console.log(`Processing batch ${i + 1}/${batches.length} (${batch.length} workflows)`);
      
      try {
        const batchResult = await this.performBulkOperation(operation, batch, options);
        results.push(batchResult);
        
        // Small delay between batches to avoid rate limits
        if (i < batches.length - 1) {
          await new Promise(resolve => setTimeout(resolve, 1000));
        }
      } catch (error) {
        console.error(`Batch ${i + 1} failed:`, error);
        results.push({ error: error.message, batch });
      }
    }
    
    return this.aggregateResults(results, operation);
  }

  createBatches(workflowIds) {
    const batches = [];
    for (let i = 0; i < workflowIds.length; i += this.batchSize) {
      batches.push(workflowIds.slice(i, i + this.batchSize));
    }
    return batches;
  }

  async performBulkOperation(operation, workflowIds, options) {
    const response = await fetch('/api/workflows', {
      method: 'PATCH',
      headers: {
        'x-api-key': this.apiKey,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operation,
        workflowIds,
        ...options
      })
    });

    if (!response.ok) {
      const error = await response.json();
      throw new Error(error.message || `Bulk ${operation} failed`);
    }

    return await response.json();
  }

  aggregateResults(batchResults, operation) {
    const aggregated = {
      operation,
      results: [],
      summary: {
        requested: 0,
        successful: 0,
        failed: 0
      }
    };

    batchResults.forEach(batch => {
      if (batch.error) {
        // Handle batch errors
        console.error('Batch error:', batch.error);
        return;
      }

      aggregated.results.push(...batch.results);
      aggregated.summary.requested += batch.summary.requested;
      aggregated.summary.successful += batch.summary.successful;
      aggregated.summary.failed += batch.summary.failed;
    });

    return aggregated;
  }
}

// Usage
const bulkManager = new BulkOperationManager('YOUR_API_KEY');

// Process 500 workflows in batches of 100
const result = await bulkManager.executeBulkOperation('deploy', largWorkflowIdList);
console.log(`Processed ${result.summary.successful} out of ${result.summary.requested} workflows`);
```

### Workflow Migration

Complete example for migrating workflows between environments:

```javascript
async function migrateWorkflows(sourceWorkspaceId, targetWorkspaceId, options = {}) {
  const {
    folderMapping = {},
    preserveDeploymentStatus = false,
    tagPrefix = 'migrated',
    dryRun = false
  } = options;

  console.log(`${dryRun ? 'Dry run: ' : ''}Migrating workflows from ${sourceWorkspaceId} to ${targetWorkspaceId}`);

  // 1. Get all workflows from source workspace
  const sourceResponse = await fetch('/api/workflows?' + new URLSearchParams({
    workspaceId: sourceWorkspaceId,
    limit: '100'
  }), {
    headers: { 'x-api-key': 'YOUR_API_KEY' }
  });

  const { data: sourceWorkflows } = await sourceResponse.json();
  console.log(`Found ${sourceWorkflows.length} workflows to migrate`);

  if (dryRun) {
    console.log('Workflows to migrate:', sourceWorkflows.map(w => w.name));
    return;
  }

  // 2. Copy workflows to target workspace
  const workflowIds = sourceWorkflows.map(w => w.id);
  const copyResult = await fetch('/api/workflows', {
    method: 'PATCH',
    headers: {
      'x-api-key': 'YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      operation: 'copy',
      workflowIds,
      targetWorkspaceId,
      targetFolderId: folderMapping.default || null,
      preserveCollaborators: false // Don't copy collaborators in migration
    })
  });

  const copyResults = await copyResult.json();
  const newWorkflowIds = copyResults.results
    .filter(r => r.status === 'copied')
    .map(r => r.newId);

  console.log(`Successfully copied ${newWorkflowIds.length} workflows`);

  // 3. Add migration tags
  if (newWorkflowIds.length > 0) {
    await fetch('/api/workflows', {
      method: 'PATCH',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operation: 'tag',
        workflowIds: newWorkflowIds,
        tags: [tagPrefix, `source-${sourceWorkspaceId}`, new Date().toISOString().split('T')[0]]
      })
    });
  }

  // 4. Deploy if preserving deployment status
  if (preserveDeploymentStatus) {
    const deployedSourceIds = sourceWorkflows
      .filter(w => w.isDeployed)
      .map(w => w.id);
    
    if (deployedSourceIds.length > 0) {
      // Find corresponding copied workflow IDs
      const deployIds = copyResults.results
        .filter(r => r.status === 'copied' && deployedSourceIds.includes(r.id))
        .map(r => r.newId);

      if (deployIds.length > 0) {
        await fetch('/api/workflows', {
          method: 'PATCH',
          headers: {
            'x-api-key': 'YOUR_API_KEY',
            'Content-Type': 'application/json'
          },
          body: JSON.stringify({
            operation: 'deploy',
            workflowIds: deployIds
          })
        });

        console.log(`Deployed ${deployIds.length} workflows to maintain status`);
      }
    }
  }

  return {
    sourceWorkflows: sourceWorkflows.length,
    copiedWorkflows: newWorkflowIds.length,
    failedCopies: copyResults.summary.failed,
    newWorkflowIds
  };
}

// Usage
const migrationResult = await migrateWorkflows('ws_dev', 'ws_staging', {
  preserveDeploymentStatus: true,
  tagPrefix: 'staging-migration',
  dryRun: false
});

console.log('Migration completed:', migrationResult);
```

## Best Practices

<Callout type="info">
**Performance Tips**
- Limit bulk operations to 100 workflows per request
- Add delays between large batch operations to avoid rate limits
- Use specific error handling for partial failures
- Always verify results before proceeding with additional operations
</Callout>

### Bulk Operation Guidelines

1. **Verify Permissions** - Ensure you have appropriate permissions for all target workflows
2. **Backup Important Data** - Export workflows before destructive operations
3. **Test with Small Batches** - Start with a few workflows to verify behavior
4. **Monitor Results** - Check the summary and individual results for each operation
5. **Handle Partial Failures** - Plan for scenarios where some workflows fail
6. **Use Dry Run Options** - When available, test operations without making changes

```javascript
// Pattern 1: Safe bulk delete with confirmation
async function safeBulkDelete(workflowIds) {
  console.log(`Preparing to delete ${workflowIds.length} workflows`);
  
  // Get workflow names for confirmation
  const workflows = await Promise.all(
    workflowIds.map(id => 
      fetch(`/api/workflows/${id}`, {
        headers: { 'x-api-key': 'YOUR_API_KEY' }
      }).then(r => r.json()).then(data => data.data)
    )
  );
  
  console.log('Workflows to delete:');
  workflows.forEach(w => console.log(`- ${w.name} (${w.runCount} runs)`));
  
  const confirmed = confirm(`Delete ${workflows.length} workflows? This cannot be undone.`);
  if (!confirmed) return;
  
  const result = await fetch('/api/workflows', {
    method: 'PATCH',
    headers: {
      'x-api-key': 'YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      operation: 'delete',
      workflowIds
    })
  });
  
  return result.json();
}

// Pattern 2: Conditional bulk operations
async function archiveInactiveWorkflows(days = 30) {
  const cutoffDate = new Date(Date.now() - days * 24 * 60 * 60 * 1000).toISOString();
  
  // Find inactive workflows
  const response = await fetch('/api/workflows?' + new URLSearchParams({
    updatedBefore: cutoffDate,
    status: 'deployed',
    maxRunCount: '0',
    limit: '100'
  }), {
    headers: { 'x-api-key': 'YOUR_API_KEY' }
  });
  
  const { data: inactiveWorkflows } = await response.json();
  
  if (inactiveWorkflows.length === 0) {
    console.log('No inactive workflows found');
    return;
  }
  
  console.log(`Found ${inactiveWorkflows.length} inactive workflows`);
  
  // Move to archive folder and undeploy
  const workflowIds = inactiveWorkflows.map(w => w.id);
  
  const [moveResult, undeployResult, tagResult] = await Promise.all([
    // Move to archive folder
    fetch('/api/workflows', {
      method: 'PATCH',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operation: 'move',
        workflowIds,
        targetFolderId: 'folder_archive'
      })
    }).then(r => r.json()),
    
    // Undeploy
    fetch('/api/workflows', {
      method: 'PATCH',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operation: 'undeploy',
        workflowIds
      })
    }).then(r => r.json()),
    
    // Add archived tag
    fetch('/api/workflows', {
      method: 'PATCH',
      headers: {
        'x-api-key': 'YOUR_API_KEY',
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        operation: 'tag',
        workflowIds,
        tags: ['archived', `archived-${new Date().toISOString().split('T')[0]}`]
      })
    }).then(r => r.json())
  ]);
  
  console.log('Archive results:', {
    moved: moveResult.summary.successful,
    undeployed: undeployResult.summary.successful,
    tagged: tagResult.summary.successful
  });
}
```
