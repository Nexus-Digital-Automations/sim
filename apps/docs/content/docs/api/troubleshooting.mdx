# API Troubleshooting

This guide provides advanced troubleshooting techniques for common API integration issues, debugging strategies, and best practices for maintaining reliable integrations with the Sim API.

## Debugging Workflow

### 1. Identify the Issue

Start by gathering information about the problem:

- **Error Messages**: Collect exact error messages and codes
- **Request Details**: Log the full request (headers, body, URL)
- **Response Details**: Capture complete response including headers
- **Environment**: Note which environment (dev, staging, prod)
- **Timing**: When did the issue start? Is it consistent?

### 2. Reproduce the Issue

```javascript
class ApiKeyManager {
  constructor() {
    this.validationCache = new Map();
    this.cacheTimeout = 5 * 60 * 1000; // 5 minutes
  }

  async getValidApiKey() {
    // Try environment variables first
    const envKey = process.env.SIM_API_KEY;
    if (await this.validateKey(envKey)) {
      return envKey;
    }

    // Try different environment variable names
    const alternateKeys = [
      process.env.SIM_API_TOKEN,
      process.env.WORKFLOW_API_KEY,
      process.env.API_KEY
    ];

    for (const key of alternateKeys) {
      if (key && await this.validateKey(key)) {
        console.log('✅ Found valid API key in alternate environment variable');
        return key;
      }
    }

    throw new Error('No valid API key found. Please check your environment variables.');
  }

  async validateKey(apiKey) {
    if (!apiKey) return false;

    // Check cache first
    const cacheKey = apiKey.slice(-8); // Last 8 chars for cache key
    const cached = this.validationCache.get(cacheKey);
    if (cached && Date.now() - cached.timestamp < this.cacheTimeout) {
      return cached.isValid;
    }

    try {
      const response = await fetch('https://api.sim.dev/api/workflows?limit=1', {
        headers: { 'x-api-key': apiKey }
      });

      const isValid = response.status !== 401;
      
      // Cache result
      this.validationCache.set(cacheKey, {
        isValid,
        timestamp: Date.now()
      });

      return isValid;
    } catch {
      return false;
    }
  }

  rotateApiKey() {
    console.log(`
```

#### Problem: "Session Expired"

```javascript
// Handle session expiration
class SessionManager {
  constructor(onSessionExpired) {
    this.onSessionExpired = onSessionExpired;
    this.isRefreshing = false;
    this.failedQueue = [];
  }

  async handleRequest(requestFn) {
    try {
      return await requestFn();
    } catch (error) {
      if (error.status === 401 && error.details?.type === 'expired_session') {
        return await this.handleSessionExpired(requestFn);
      }
      throw error;
    }
  }

  async handleSessionExpired(requestFn) {
    if (this.isRefreshing) {
      // Queue the request while refreshing
      return new Promise((resolve, reject) => {
        this.failedQueue.push({ resolve, reject, requestFn });
      });
    }

    this.isRefreshing = true;

    try {
      // Attempt to refresh session
      await this.refreshSession();
      
      // Retry original request
      const result = await requestFn();
      
      // Process queued requests
      this.processQueue(null);
      
      return result;
    } catch (refreshError) {
      this.processQueue(refreshError);
      
      // Redirect to login or show auth modal
      if (this.onSessionExpired) {
        this.onSessionExpired();
      }
      
      throw refreshError;
    } finally {
      this.isRefreshing = false;
    }
  }

  async refreshSession() {
    const response = await fetch('/api/auth/refresh', {
      method: 'POST',
      credentials: 'include'
    });

    if (!response.ok) {
      throw new Error('Session refresh failed');
    }
  }

  processQueue(error) {
    this.failedQueue.forEach(({ resolve, reject, requestFn }) => {
      if (error) {
        reject(error);
      } else {
        resolve(requestFn());
      }
    });
    
    this.failedQueue = [];
  }
}

// Usage
const sessionManager = new SessionManager(() => {
  // Redirect to login
  window.location.href = '/login';
});

// Wrap API calls
async function apiCall() {
  return await sessionManager.handleRequest(async () => {
    const response = await fetch('/api/workflows', {
      credentials: 'include'
    });
    
    if (!response.ok) {
      const error = await response.json();
      const apiError = new Error(error.message);
      apiError.status = response.status;
      apiError.details = error.details;
      throw apiError;
    }
    
    return await response.json();
  });
}
```

```javascript
// Detect and analyze rate limiting
class RateLimitMonitor {
  constructor() {
    this.requestTimes = [];
    this.rateLimitHits = [];
  }

  recordRequest() {
    const now = Date.now();
    this.requestTimes.push(now);
    
    // Keep only last hour of requests
    const oneHour = 60 * 60 * 1000;
    this.requestTimes = this.requestTimes.filter(time => now - time < oneHour);
  }

  recordRateLimit(retryAfter) {
    this.rateLimitHits.push({
      timestamp: Date.now(),
      retryAfter
    });
  }

  analyzeUsage() {
    const now = Date.now();
    const oneMinute = 60 * 1000;
    const oneHour = 60 * 60 * 1000;

    const requestsLastMinute = this.requestTimes.filter(time => 
      now - time < oneMinute
    ).length;

    const requestsLastHour = this.requestTimes.filter(time => 
      now - time < oneHour
    ).length;

    const recentRateLimits = this.rateLimitHits.filter(hit => 
      now - hit.timestamp < oneHour
    ).length;

    return {
      requestsLastMinute,
      requestsLastHour,
      recentRateLimits,
      avgRequestsPerMinute: requestsLastHour / 60,
      isApproachingLimit: requestsLastMinute > 50 // Assuming 60/min limit
    };
  }

  getRecommendations() {
    const analysis = this.analyzeUsage();
    const recommendations = [];

    if (analysis.isApproachingLimit) {
      recommendations.push('Slow down requests to avoid rate limiting');
    }

    if (analysis.recentRateLimits > 0) {
      recommendations.push('Implement exponential backoff');
      recommendations.push('Consider batching operations');
    }

    if (analysis.avgRequestsPerMinute > 30) {
      recommendations.push('Consider upgrading your plan for higher limits');
    }

    return recommendations;
  }
}

// Usage
const monitor = new RateLimitMonitor();

async function monitoredRequest(url, options) {
  monitor.recordRequest();
  
  try {
    const response = await fetch(url, options);
    
    // Check rate limit headers
    const remaining = parseInt(response.headers.get('x-ratelimit-remaining') || '0');
    const limit = parseInt(response.headers.get('x-ratelimit-limit') || '0');
    
    if (remaining < limit * 0.1) { // Less than 10% remaining

      throw new Error(`Rate limited. Retry after ${retryAfter}s`);
    }
    
    return response;
  } catch (error) {
    const analysis = monitor.analyzeUsage();
    const recommendations = monitor.getRecommendations();
    
    if (recommendations.length > 0) {
      console.warn('Rate limit recommendations:', recommendations);
    }
    
    throw error;
  }
}
```

```javascript
// Robust rate limit handling
class RateLimitHandler {
  constructor(options = {}) {
    this.baseDelay = options.baseDelay || 1000;
    this.maxDelay = options.maxDelay || 60000;
    this.maxRetries = options.maxRetries || 5;
    this.jitterFactor = options.jitterFactor || 0.1;
  }

  async withRateLimit(requestFn, options = {}) {
    let lastError;
    
    for (let attempt = 0; attempt < this.maxRetries; attempt++) {
      try {
        return await requestFn();
      } catch (error) {
        lastError = error;
        
        // Only retry on rate limit errors
        if (error.status !== 429) {
          throw error;
        }
        
        if (attempt === this.maxRetries - 1) {
          throw new Error(`Rate limit retries exhausted. Last error: ${error.message}`);
        }
        
        const delay = this.calculateDelay(attempt, error);
        console.log(`Rate limited. Waiting ${delay}ms before retry ${attempt + 1}...`);
        
        await this.sleep(delay);
      }
    }
    
    throw lastError;
  }

  calculateDelay(attempt, error) {
    // Use server-provided retry-after if available
    if (error.retryAfter) {
      return Math.min(error.retryAfter * 1000, this.maxDelay);
    }
    
    // Exponential backoff with jitter
    const exponentialDelay = this.baseDelay * Math.pow(2, attempt);
    const jitter = exponentialDelay * this.jitterFactor * Math.random();
    
    return Math.min(exponentialDelay + jitter, this.maxDelay);
  }

  sleep(ms) {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}

// Request queue to prevent overwhelming the API
class RequestQueue {
  constructor(concurrency = 5, delayBetweenRequests = 200) {
    this.concurrency = concurrency;
    this.delay = delayBetweenRequests;
    this.queue = [];
    this.running = 0;
    this.rateLimitHandler = new RateLimitHandler();
  }

  async add(requestFn) {
    return new Promise((resolve, reject) => {
      this.queue.push({
        requestFn,
        resolve,
        reject
      });
      
      this.process();
    });
  }

  async process() {
    if (this.running >= this.concurrency || this.queue.length === 0) {
      return;
    }

    this.running++;
    const { requestFn, resolve, reject } = this.queue.shift();

    try {
      const result = await this.rateLimitHandler.withRateLimit(requestFn);
      resolve(result);
    } catch (error) {
      reject(error);
    } finally {
      this.running--;
      
      // Add delay between requests
      if (this.queue.length > 0) {
        setTimeout(() => this.process(), this.delay);
      }
    }
  }
}

// Usage
const requestQueue = new RequestQueue(3, 500); // 3 concurrent, 500ms between

        throw apiError;
      }

      return await response.json();
    });
    
    results.push(result);
  }
  
  return results;
}
```

```javascript
// Prevent rate limiting through intelligent request management
class IntelligentApiClient {
  constructor(apiKey) {
    this.apiKey = apiKey;
    this.requestStats = {
      count: 0,
      lastReset: Date.now(),
      windowSize: 60000 // 1 minute
    };
    this.adaptiveDelay = 100; // Start with 100ms delay
    this.maxDelay = 2000; // Max 2s delay
  }

  async request(endpoint, options = {}) {
    // Check if we should slow down
    await this.adaptiveThrottle();
    
    try {
      const response = await fetch(endpoint, {
        ...options,
        headers: {
          'x-api-key': this.apiKey,
          ...options.headers
        }
      });

      this.updateStats(response);
      
      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }
      
      return await response.json();
    } catch (error) {
      this.handleError(error);
      throw error;
    }
  }

  async adaptiveThrottle() {
    // Reset stats if window expired
    const now = Date.now();
    if (now - this.requestStats.lastReset > this.requestStats.windowSize) {
      this.requestStats.count = 0;
      this.requestStats.lastReset = now;
    }

    // Increase delay if making too many requests
    const requestsPerMinute = this.requestStats.count;
    if (requestsPerMinute > 50) {
      this.adaptiveDelay = Math.min(this.adaptiveDelay * 1.5, this.maxDelay);
    } else if (requestsPerMinute < 20) {
      this.adaptiveDelay = Math.max(this.adaptiveDelay * 0.9, 100);
    }

    if (this.adaptiveDelay > 100) {
      await new Promise(resolve => setTimeout(resolve, this.adaptiveDelay));
    }
  }

  updateStats(response) {
    this.requestStats.count++;

    // Adjust delay based on rate limit headers
    const remaining = parseInt(response.headers.get('x-ratelimit-remaining') || '1000');
    const limit = parseInt(response.headers.get('x-ratelimit-limit') || '1000');
    const resetTime = parseInt(response.headers.get('x-ratelimit-reset') || '0');

    // If we're using too much quota, slow down
    const usagePercentage = ((limit - remaining) / limit) * 100;
    
    if (usagePercentage > 80) {
      this.adaptiveDelay = Math.min(this.adaptiveDelay * 2, this.maxDelay);
    } else if (usagePercentage < 50) {
      this.adaptiveDelay = Math.max(this.adaptiveDelay * 0.8, 100);
    }
  }

  handleError(error) {
    if (error.message.includes('429')) {
      // Significantly increase delay after rate limiting
      this.adaptiveDelay = Math.min(this.adaptiveDelay * 3, this.maxDelay);
    }
  }

  // Batch operations to reduce API calls
  async batchOperation(operation, items, batchSize = 10) {
    const results = [];
    
    for (let i = 0; i < items.length; i += batchSize) {
      const batch = items.slice(i, i + batchSize);
      console.log(`Processing batch ${Math.floor(i / batchSize) + 1}/${Math.ceil(items.length / batchSize)}`);
      
      const batchResults = await Promise.all(
        batch.map(item => operation(item))
      );
      
      results.push(...batchResults);
      
      // Add delay between batches
      if (i + batchSize < items.length) {
        await new Promise(resolve => setTimeout(resolve, this.adaptiveDelay));
      }
    }
    
    return results;
  }
}
```

```javascript
      try {
        const response = await fetch(url, {
          ...options,
          signal: controller.signal
        });

        clearTimeout(timeoutId);
        return response;

      } catch (error) {
        lastError = error;
        
        const errorType = this.classifyError(error);
        console.log(`Attempt ${attempt + 1} failed: ${errorType}`);

        // Don't retry non-network errors
        if (!this.retryableErrors.includes(errorType)) {
          break;
        }

        // Don't retry on last attempt
        if (attempt === this.retryDelays.length) {
          break;
        }

        // Wait before retry
        const delay = this.retryDelays[attempt];
        console.log(`Retrying in ${delay}ms...`);
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }

    clearTimeout(timeoutId);
    throw new Error(`Network request failed after ${this.retryDelays.length + 1} attempts: ${lastError.message}`);
  }

  classifyError(error) {
    if (error.name === 'AbortError') {
      return 'TIMEOUT';
    }
    
    if (error.message.includes('Failed to fetch')) {
      return 'NETWORK_ERROR';
    }
    
    if (error.message.includes('ERR_NETWORK_CHANGED')) {
      return 'NETWORK_ERROR';
    }
    
    if (error.message.includes('ERR_INTERNET_DISCONNECTED')) {
      return 'NETWORK_ERROR';
    }
    
    return 'UNKNOWN_ERROR';
  }

  async checkConnectivity() {
    const tests = [
      { name: 'DNS Resolution', test: () => fetch('https://1.1.1.1', { mode: 'no-cors' }) },
      { name: 'Internet Connectivity', test: () => fetch('https://httpbin.org/get', { mode: 'cors' }) },
      { name: 'Sim API Reachability', test: () => fetch('https://api.sim.dev/health', { mode: 'cors' }) }
    ];

    const results = [];

    for (const { name, test } of tests) {
      try {
        const start = Date.now();
        await test();
        const duration = Date.now() - start;
        results.push({ name, status: 'OK', duration });
        console.log(`✅ ${name}: OK (${duration}ms)`);
      } catch (error) {
        results.push({ name, status: 'FAILED', error: error.message });
        console.log(`❌ ${name}: FAILED - ${error.message}`);
      }
    }

    return results;
  }
}

  retryDelays: [1000, 3000, 5000]
});
```

```javascript
      console.log(`   ${checkpoint.name}: ${duration.toFixed(2)}ms`);
    });

    // Performance warnings
    if (totalTime > 5000) {
      console.warn(`⚠️  Slow request detected: ${totalTime.toFixed(2)}ms`);

💡 Performance Optimization Suggestions:
   - Use pagination to reduce response size
   - Add specific field filters to reduce payload
   - Implement client-side caching
   - Consider using bulk operations
   - Check network connectivity and latency
    `);
  }

      slowRequests,
      slowRequestPercentage: (slowRequests / metrics.length) * 100
    };
  }
}
```

```javascript
// Intelligent caching for API responses
class SmartCache {
  constructor(options = {}) {
    this.cache = new Map();
    this.defaultTtl = options.defaultTtl || 5 * 60 * 1000; // 5 minutes
    this.maxSize = options.maxSize || 1000;
    this.hitCount = 0;
    this.missCount = 0;
  }

  // Generate cache key from request details
  generateKey(method, url, headers = {}, body = null) {
    const keyData = {
      method: method.toUpperCase(),
      url,
      relevantHeaders: {
        'x-api-key': headers['x-api-key']?.slice(-8), // Last 8 chars only
      },
      body: body ? JSON.stringify(body) : null
    };

    return btoa(JSON.stringify(keyData));
  }

  // Check if request is cacheable
  isCacheable(method, url, headers = {}) {
    // Only cache GET requests
    if (method.toUpperCase() !== 'GET') {
      return false;
    }

    // Don't cache real-time or user-specific data
    const noCachePatterns = [
      '/status',
      '/logs',
      '/presence',
      '/activity',
      '/realtime'
    ];

    return !noCachePatterns.some(pattern => url.includes(pattern));
  }

  // Get cache entry
  get(key) {
    const entry = this.cache.get(key);
    
    if (!entry) {
      this.missCount++;
      return null;
    }

    // Check if expired
    if (Date.now() > entry.expiry) {
      this.cache.delete(key);
      this.missCount++;
      return null;
    }

    this.hitCount++;
    entry.lastAccessed = Date.now();
    return entry.data;
  }

  // Set cache entry
  set(key, data, ttl = null) {
    // Enforce max size
    if (this.cache.size >= this.maxSize) {
      this.evictOldest();
    }

    const entry = {
      data,
      expiry: Date.now() + (ttl || this.defaultTtl),
      created: Date.now(),
      lastAccessed: Date.now(),
      accessCount: 1
    };

    this.cache.set(key, entry);
  }

  // Evict oldest entries
  evictOldest() {
    let oldest = null;
    let oldestTime = Date.now();

    for (const [key, entry] of this.cache.entries()) {
      if (entry.lastAccessed < oldestTime) {
        oldest = key;
        oldestTime = entry.lastAccessed;
      }
    }

    if (oldest) {
      this.cache.delete(oldest);
    }
  }

  // Invalidate cache entries by pattern
  invalidate(pattern) {
    const regex = new RegExp(pattern);
    const keysToDelete = [];

    for (const key of this.cache.keys()) {
      if (regex.test(key)) {
        keysToDelete.push(key);
      }
    }

    keysToDelete.forEach(key => this.cache.delete(key));
    console.log(`Invalidated ${keysToDelete.length} cache entries matching ${pattern}`);
  }

  // Get cache statistics
  getStats() {
    const total = this.hitCount + this.missCount;
    const hitRate = total > 0 ? (this.hitCount / total) * 100 : 0;

    return {
      size: this.cache.size,
      maxSize: this.maxSize,
      hitCount: this.hitCount,
      missCount: this.missCount,
      hitRate: hitRate.toFixed(2) + '%',
      efficiency: hitRate > 50 ? 'Good' : hitRate > 20 ? 'Fair' : 'Poor'
    };
  }
}

// Cached API client
class CachedApiClient {
  constructor(apiKey, cacheOptions = {}) {
    this.apiKey = apiKey;
    this.cache = new SmartCache(cacheOptions);
  }

  async request(method, endpoint, options = {}) {
    const url = `https://api.sim.dev${endpoint}`;
    const headers = {
      'x-api-key': this.apiKey,
      'Content-Type': 'application/json',
      ...options.headers
    };

    const cacheKey = this.cache.generateKey(method, url, headers, options.body);

    // Try cache first for GET requests
    if (this.cache.isCacheable(method, url)) {
      const cached = this.cache.get(cacheKey);
      if (cached) {
        console.log('📦 Cache hit');
        return cached;
      }
    }

    console.log('🌐 Making API request');
    
    try {
      const response = await fetch(url, {
        method,
        headers,
        ...options
      });

      if (!response.ok) {
        throw new Error(`HTTP ${response.status}: ${response.statusText}`);
      }

      const data = await response.json();

      // Cache successful GET responses
      if (this.cache.isCacheable(method, url)) {
        const ttl = this.determineCacheTtl(endpoint);
        this.cache.set(cacheKey, data, ttl);
      }

      // Invalidate related cache entries for write operations
      if (method !== 'GET') {
        this.invalidateRelatedCache(endpoint);
      }

      return data;
    } catch (error) {
      console.error('API request failed:', error);
      throw error;
    }
  }

  determineCacheTtl(endpoint) {
    // Different TTLs for different types of data
    const ttlMap = {
      '/api/workflows': 2 * 60 * 1000,      // 2 minutes
      '/api/templates': 10 * 60 * 1000,     // 10 minutes  
      '/api/users': 5 * 60 * 1000,          // 5 minutes
      '/api/workspaces': 10 * 60 * 1000     // 10 minutes
    };

    for (const [pattern, ttl] of Object.entries(ttlMap)) {
      if (endpoint.includes(pattern)) {
        return ttl;
      }
    }

    return this.cache.defaultTtl;
  }

  invalidateRelatedCache(endpoint) {
    // Invalidate related cache entries when data changes
    const invalidationMap = {
      '/api/workflows': ['workflows'],
      '/api/templates': ['templates', 'workflows'],
      '/api/workspaces': ['workspaces', 'workflows']
    };

    for (const [pattern, cachePatterns] of Object.entries(invalidationMap)) {
      if (endpoint.includes(pattern)) {
        cachePatterns.forEach(cachePattern => {
          this.cache.invalidate(cachePattern);
        });
        break;
      }
    }
  }

  getCacheStats() {
    return this.cache.getStats();
  }

  clearCache() {
    this.cache = new SmartCache();
    console.log('📦 Cache cleared');
  }
}

// Usage
```

```javascript
// Production monitoring for API integration
class ApiHealthMonitor {
  constructor(apiKey, options = {}) {
    this.apiKey = apiKey;
    this.interval = options.interval || 60000; // 1 minute
    this.alertThreshold = options.alertThreshold || 5; // 5 failures
    this.endpoints = options.endpoints || ['/api/workflows'];
    
    this.failures = new Map();
    this.isMonitoring = false;
    this.onAlert = options.onAlert || console.error;
  }

  start() {
    if (this.isMonitoring) return;
    
    this.isMonitoring = true;
    console.log('🔍 Starting API health monitoring...');
    
    this.monitoringInterval = setInterval(() => {
      this.checkHealth();
    }, this.interval);
    
    // Initial check
    this.checkHealth();
  }

  stop() {
    if (!this.isMonitoring) return;
    
    this.isMonitoring = false;
    clearInterval(this.monitoringInterval);
    console.log('⏹️ API health monitoring stopped');
  }

  async checkHealth() {
    const results = await Promise.all(
      this.endpoints.map(endpoint => this.checkEndpoint(endpoint))
    );

    const failedEndpoints = results.filter(r => !r.healthy);
    
    if (failedEndpoints.length > 0) {
      this.handleFailures(failedEndpoints);
    } else {
      // Reset failure counts on success
      this.failures.clear();
    }

    this.logHealthStatus(results);
  }

  async checkEndpoint(endpoint) {
    const start = Date.now();
    
    try {
      const response = await fetch(`https://api.sim.dev${endpoint}?limit=1`, {
        headers: { 'x-api-key': this.apiKey },
        timeout: 10000
      });

      const duration = Date.now() - start;
      const healthy = response.ok && duration < 5000; // Must be OK and under 5s

      return {
        endpoint,
        healthy,
        status: response.status,
        duration,
        timestamp: new Date().toISOString()
      };
    } catch (error) {
      return {
        endpoint,
        healthy: false,
        status: 0,
        duration: Date.now() - start,
        error: error.message,
        timestamp: new Date().toISOString()
      };
    }
  }

  handleFailures(failedEndpoints) {
    failedEndpoints.forEach(failure => {
      const count = (this.failures.get(failure.endpoint) || 0) + 1;
      this.failures.set(failure.endpoint, count);

      if (count >= this.alertThreshold) {
        this.triggerAlert(failure, count);
      }
    });
  }

  triggerAlert(failure, count) {
    const alert = {
      type: 'API_HEALTH_ALERT',
      endpoint: failure.endpoint,
      failureCount: count,
      lastError: failure.error,
      status: failure.status,
      timestamp: failure.timestamp,
      severity: count >= 10 ? 'CRITICAL' : 'WARNING'
    };

    console.error(`🚨 API Health Alert:`, alert);
    this.onAlert(alert);
  }

  logHealthStatus(results) {
    const healthy = results.filter(r => r.healthy).length;
    const total = results.length;
    
    const avgDuration = results.reduce((sum, r) => sum + r.duration, 0) / total;
    
    console.log(`💓 API Health: ${healthy}/${total} endpoints healthy (avg ${avgDuration.toFixed(0)}ms)`);
    
    results.forEach(result => {
      const icon = result.healthy ? '✅' : '❌';
      const duration = `${result.duration}ms`;
      const status = result.status || 'ERROR';
      
      console.log(`  ${icon} ${result.endpoint}: ${status} (${duration})`);
    });
  }

  getHealthReport() {
    return {
      monitoring: this.isMonitoring,
      failures: Object.fromEntries(this.failures),
      lastCheck: new Date().toISOString()
    };
  }
}
```
